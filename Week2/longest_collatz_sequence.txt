#include <bits/stdc++.h>

using namespace std;

void print_coll(long long N) {
  while (true) {
    
    if (N == 1) {
      printf("%d", N);
      break;
    }
    
    printf("%d ", N);
   	N = (N % 2 == 0) ? (N / 2) : ((N * 3) + 1);
    
  }
  
}

long long leng_coll(long long N) {
  long long cnt = 0;
  
  while (true) {
    if (N == 1)
      break;
    
    N = (N % 2 == 0) ? (N / 2) : ((N * 3) + 1);
    cnt++;
  }
    
  return cnt;
}
  
int main() {
  long long a, b;
  long long max_num = 0, max_length = 0; // 가장 긴 수, 길이
  
  scanf("%lld %lld", &a, &b);
  
  for (int i = a; i <= b; i++) {
    long long temp = leng_coll(i);
    if (temp >= max_length) {
      max_num = i;
      max_length = temp;
    }
  
  }
  
  printf("%lld %lld\n", max_num, max_length);
  print_coll(max_num);
  
  return 0;
  
}
